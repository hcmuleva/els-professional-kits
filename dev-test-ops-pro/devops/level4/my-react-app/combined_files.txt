

--- FILE: ./Chart.yaml ---

apiVersion: v2
name: my-react-app
description: A Helm chart to deploy a React + Vite app with Nginx
type: application
version: 0.1.0
appVersion: "1.0.0"


--- FILE: ./templates/_helpers.tpl ---

{{/*
Return the fully qualified name of the chart.
*/}}
{{- define "my-react-app.fullname" -}}
{{- printf "%s-%s" .Release.Name .Chart.Name | trunc 63 | trimSuffix "-" -}}
{{- end -}}

{{/*
Return chart name
*/}}
{{- define "my-react-app.name" -}}
{{- .Chart.Name -}}
{{- end -}}


--- FILE: ./templates/deployment.yaml ---

apiVersion: apps/v1
kind: Deployment
metadata:
  name: {{ include "my-react-app.fullname" . }}
  labels:
    app: {{ include "my-react-app.name" . }}
spec:
  replicas: {{ .Values.replicaCount }}
  selector:
    matchLabels:
      app: {{ include "my-react-app.name" . }}
  template:
    metadata:
      labels:
        app: {{ include "my-react-app.name" . }}
    spec:
      containers:
        - name: react-app
          image: "{{ .Values.image.repository }}:{{ .Values.image.tag }}"
          imagePullPolicy: {{ .Values.image.pullPolicy }}
          ports:
            - containerPort: 80
          readinessProbe:
            httpGet:
              path: /
              port: 80
            initialDelaySeconds: 5
            periodSeconds: 10


--- FILE: ./templates/ingress.yaml ---

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: {{ include "my-react-app.fullname" . }}
  labels:
    app: {{ include "my-react-app.name" . }}
spec:
  ingressClassName: nginx
  rules:
    - host: {{ .Values.ingress.host }}
      http:
        paths:
          - path: /ui
            pathType: Prefix
            backend:
              service:
                name: {{ include "my-react-app.fullname" . }}
                port:
                  number: 80


--- FILE: ./templates/service.yaml ---

apiVersion: v1
kind: Service
metadata:
  name: {{ include "my-react-app.fullname" . }}
spec:
  type: {{ .Values.service.type }}
  selector:
    app: {{ include "my-react-app.name" . }}
  ports:
    - port: {{ .Values.service.port }}
      targetPort: 80
      protocol: TCP
      name: http


--- FILE: ./values.yaml ---

replicaCount: 1

image:
  repository: social-network
  tag: latest
  pullPolicy: IfNotPresent

service:
  type: ClusterIP
  port: 80

ingress:
  enabled: true
  className: nginx
  annotations:
    nginx.ingress.kubernetes.io/rewrite-target: /$1
  hosts:
    - host: yourdomain.com
      paths:
        - path: /ui/?(.*)
          pathType: Prefix
  tls: []
